def job_unread_status(job)
  status = job.is_unread?(@char) ? t('jobs.unread_marker') : ""

def page_marker
  page_marker = t('pages.page_x_of_y', :x => @current_page, :y => @total_pages)
  "%x!#{page_marker.center(78, '-')}%xn"
